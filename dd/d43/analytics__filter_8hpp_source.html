27 static void ToPoints(models::Detections &detections,
28  const std::vector<va::Object> &tracks, std::vector<influxdb::Point> &points) {
29  points.clear();
30  points.push_back(
31  influxdb::Point("frame")
32  .Field("id", detections.id)
33  .Time(detections.timestamp)
34  );
35 
36  for (auto &detection : detections.data) {
37  points.push_back(
38  influxdb::Point("detection")
39  .Field("id", detection.id)
40  .Field("det_id", detection.det_id)
41  .Tag("label", detection.label)
42  .Field("frame", detections.id)
43  .Time(detections.timestamp)
44  );
45  }
46 
47  for (auto &track : tracks) {
48  auto p = influxdb::Point("track")
49  .Field("id", track.id)
50  .Field("det_id", track.det_id)
51  .Field("frame", detections.id)
52  .Field("conf", track.conf)
53  .Field("x", track.box.x)
54  .Field("y", track.box.y)
55  .Field("width", track.box.width)
56  .Field("height", track.box.height)
57  .Tag("label", track.label)
58  .Tag("loitering", track.loitering ? "true" : "false")
59  .Tag("motionless", track.motionless ? "true" : "false")
60  .Tag("intruding", track.intruding ? "true" : "false")
61  .Time(detections.timestamp);
62 
63  switch (track.zone) {
64  case va::zone::WARNING: {
65  p.Tag("zone", "warning");
66  }
67  case va::zone::EXCLUSION: {
68  p.Tag("zone", "exclusion");
69  }
70  default: {
71  p.Tag("zone", "inclusion");
72  }
73  }
74  points.push_back(p);
75  }
76 }
