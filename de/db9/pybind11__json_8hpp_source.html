24  inline py::object from_json(const nl::json& j)
25  {
26  if (j.is_null())
27  {
28  return py::none();
29  }
30  else if (j.is_boolean())
31  {
32  return py::bool_(j.get<bool>());
33  }
34  else if (j.is_number_integer())
35  {
36  return py::int_(j.get<long>());
37  }
38  else if (j.is_number_float())
39  {
40  return py::float_(j.get<double>());
41  }
42  else if (j.is_string())
43  {
44  return py::str(j.get<std::string>());
45  }
46  else if (j.is_array())
47  {
48  py::list obj;
49  for (const auto& el : j)
50  {
51  obj.append(from_json(el));
52  }
53  return obj;
54  }
55  else // Object
56  {
57  py::dict obj;
58  for (nl::json::const_iterator it = j.cbegin(); it != j.cend(); ++it)
59  {
60  obj[py::str(it.key())] = from_json(it.value());
61  }
62  return obj;
63  }
64  }

66  inline nl::json to_json(const py::handle& obj)
67  {
68  if (obj.ptr() == nullptr || obj.is_none())
69  {
70  return nullptr;
71  }
72  if (py::isinstance<py::bool_>(obj))
73  {
74  return obj.cast<bool>();
75  }
76  if (py::isinstance<py::int_>(obj))
77  {
78  return obj.cast<long>();
79  }
80  if (py::isinstance<py::float_>(obj))
81  {
82  return obj.cast<double>();
83  }
84  if (py::isinstance<py::bytes>(obj))
85  {
86  py::module base64 = py::module::import("base64");
87  return base64.attr("b64encode")(obj).attr("decode")("utf-8").cast<std::string>();
88  }
89  if (py::isinstance<py::str>(obj))
90  {
91  return obj.cast<std::string>();
92  }
93  if (py::isinstance<py::tuple>(obj) || py::isinstance<py::list>(obj))
94  {
95  auto out = nl::json::array();
96  for (const py::handle& value : obj)
97  {
98  out.push_back(to_json(value));
99  }
100  return out;
101  }
102  if (py::isinstance<py::dict>(obj))
103  {
104  auto out = nl::json::object();
105  for (const py::handle& key : obj)
106  {
107  out[py::str(key).cast<std::string>()] = to_json(obj[key]);
108  }
109  return out;
110  }
111  throw std::runtime_error("to_json not implemented for this type of object: " + py::repr(obj).cast<std::string>());
112  }
113 }
