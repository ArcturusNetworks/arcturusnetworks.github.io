26 // Multi-producer multi-consumer concurrent queue with optional fixed size (to
27 // prevent a faster producer from bloating memory).
28 //
29 // Blocking interface:
30 // The interface producers and consumers should endeavour to use is the
31 // blocking Push()/Pop() methods. A wrapped (not inherited, since inheriting
32 // std::queue exposes thread unsafe methods) queue is protected by a queueMutex.
33 // Producers wait until there is space in the queue, consumers wait until there
34 // is data in the queue. The condition variable logic is nice and symmetric.
35 //
36 // Non-blocking interface:
37 // In more complex scenarios, a producer may want to insert into a queue if
38 // it's not full, else insert into another one. Similarly, consumers may want
39 // to dequeue if an item is immediately available, but not block and wait. For
40 // these cases, we provide the non-blocking TryPush()/TryPop() methods, which
41 // don't wait before acquiring the queue's queueMutex and report whether the
42 // operation could be completed.
43 //
44 // Stopping consumers:
45 // To stop the consumer(s), the producer(s) should enqueue EOF messages. For
46 // pointers, this may be nullptr; for user-defined types this may be some
47 // special instance (like a sentinel). Consumers should not have to check some
48 // global bool to determine whether or not to stop.
49 template <typename ElementType>
50 class ConcurrentQueue {
51  public:
52  ConcurrentQueue(const uint64_t maxSize = 0) : maxSize(maxSize) {}
53 
54  void Push(ElementType src);
55  void Pop(ElementType &dest);

58  bool PushFor(ElementType src, const std::chrono::duration<Rep, Period>& relTime);
59  template <typename Rep, typename Period>
60  bool PopFor(ElementType &dest, const std::chrono::duration<Rep, Period>& relTime);
61 
62  bool TryPush(ElementType src);
63  bool TryPop(ElementType &dest);
64 
65  std::uint64_t Size();

67  private:
68  const uint64_t maxSize;
69  std::deque<ElementType> queue;

73 };

80  if (maxSize > 0)
