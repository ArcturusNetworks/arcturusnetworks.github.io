64 static std::unique_ptr<dv::objdet::Detector> CreateObjdet(const json &config) {
65  auto model = config.at("model").get<std::string>();
66  if (dvconfig::ObjdetModelMap.find(model) == dvconfig::ObjdetModelMap.end())
67  throw std::invalid_argument("Invalid model " + model + " given");
68 
69  auto modelFile = config.at("modelFile").get<std::string>();
70 
71  auto backend = config.at("backend").get<std::string>();
72  if (dvconfig::BackendMap.find(backend) == dvconfig::BackendMap.end())
73  throw std::invalid_argument("Invalid backend " + backend + " given");
74 
75  std::string labels = "COCO_ALL_CLASSES";
76  if (config.count("labels"))
77  labels = config.at("labels").get<std::string>();
78  if (dvconfig::LabelsMap.find(labels) == dvconfig::LabelsMap.end())
79  throw std::invalid_argument("Invalid labels " + labels + " given");
80 
81  float minConf = 0.5;
82  if (config.count("minConf"))
83  minConf = config.at("minConf").get<float>();
84 
85  return dv::objdet::Create(
86  ObjdetModelMap.at(model),
87  modelFile,
88  BackendMap.at(backend),
89  LabelsMap.at(labels),
90  minConf
91  );
92 }
