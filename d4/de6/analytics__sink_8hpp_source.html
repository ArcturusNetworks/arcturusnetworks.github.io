31 static void ToPoints(models::Detections &detections, std::vector<influxdb::Point> &points) {
32  points.clear();
33  points.push_back(
34  influxdb::Point("frame")
35  .Field("id", detections.id)
36  .Time(detections.timestamp)
37  );
38 
39  for (auto &detection : detections.data) {
40  points.push_back(
41  influxdb::Point("detection")
42  .Field("id", detection.id)
43  .Field("conf", detection.conf)
44  .Field("x", detection.bbox.x)
45  .Field("y", detection.bbox.y)
46  .Field("width", detection.bbox.width)
47  .Field("height", detection.bbox.height)
48  .Tag("label", detection.label)

51  );

55 static void TagPersonFromId(const std::string name, const std::vector<uint64_t> &ids, std::vector<influxdb::Point> &points) {
56  points[0].Field(name, ids.size());
57  for (auto &point : points) {
58  if (point.GetMeasurement() != "detection") continue;
59  if (point.GetTag("label") != "person") continue;
60  point.Tag(name, "false");
61  for (auto &id : ids) {
62  if (id == point.GetField<uint64_t>("id")) {
63  point.Tag(name, "true");
64  }
65  }
66  }
67 }
