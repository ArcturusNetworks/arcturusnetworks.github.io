Detailed Description

Node which consumes data.

This class is responsible for receiving data from some IncomingStreams and consuming the data in some way (eg. displaying frames on a live stream).

The following code sample demonstrates how to implement a Sink:

namespace sp = streamproc;
namespace myproject {
class MySink : public sp::Sink {
public:
MySink(
const json &incomingData1,
const json &incomingData2
):
// Initialize base class
// Give it a unique name which will be used to create a file log
// File log can be found at: /var/log/streamproc/MySink.log
sp::Sink("MySink") {
// Initialize incoming and outgoing streams by providing configuration to filter base class
// Incoming and outgoing streams can be accessed by the key provided to the create function
CreateIncoming("data1", incomingData1);
CreateIncoming("data2", incomingData2);
// Sink can only receive data it cannot send
// CreateOutgoing(...) is not possible
}
protected:
void Process() override {
zmq::message_t message;
DataType1 data1;
incomingStreams["data1"]->Process(message);
Decode(data1, message);
logger->debug("Received: {}", data1.Str());
if (someError) {
// node failed
// exit worker routine
throw sp::error("MySink", "encountered some error");
}
incomingStreams["data2"]->Process(message);
Decode(data2, message);
logger->debug("Received: {}", data2.Str());
// consume data1 and data2
if (finisheCondition) {
logger->info("Finish condition, exiting");
throw sp::error("MySink", "finished");
}
}
};
} // namespace myproject
